Step 1: Open PowerShell as Administrator

  1. Click on the "Start" button.
  2. Type "PowerShell" in the search bar.
  3. Right-click on "Windows PowerShell" or "PowerShell."
  4. Choose "Run as administrator."

Step 2: Copy and Paste Script in PowerShell

  1. Highlight all of the script.
    2. 
function Check-UrlSecurity {
    param (
        [string]$url
    )

    try {
        $uri = [System.Uri]$url

        if ($uri.Scheme -eq "https") {
            $request = [System.Net.HttpWebRequest]::Create($url)
            $response = $request.GetResponse()

            Write-Host "The site is secured (HTTPS)."
        } elseif ($uri.Scheme -eq "http") {
            Write-Host "The site is not secured (HTTP)."
        } else {
            Write-Host "Invalid URL scheme. Please enter a valid URL starting with 'http://' or 'https://'."
        }
    } catch {
        Write-Host "Error occurred while checking the URL."
    }
}

# Prompt the user for a URL
$url = Read-Host "Enter the URL to check"

# Call the function to check the URL security
Check-UrlSecurity -url $url
